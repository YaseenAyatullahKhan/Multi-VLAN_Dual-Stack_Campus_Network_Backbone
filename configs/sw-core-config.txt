# 0. Enable IPv6
sdm prefer dual-ipv4-and-ipv6 default
exit

configure terminal

# 1. Switch Administration and Management SVI
hostname SW-Core
service password-encryption
enable secret Corp-SW-Core
ip domain-name enterprise.net
crypto key generate rsa general-keys modulus 1024
username net_admin privilege 15 secret Net_Admin@Corp-SW-Core
ip ssh version 2

line vty 0 15
 transport input ssh
 login local

# 2. Database VLAN Declarations
vlan 10
 name Corp_Operations
vlan 20
 name RD_Engineering
vlan 99
 name Native_Management
exit

# 3. Configure VLAN 99 Management SVI (Dual-Stack)
interface vlan 99
 ip address 192.168.99.10 255.255.255.0
 ipv6 address 2001:db8:acad:99::10/64
 ipv6 address fe80::10 link-local
 no shutdown
exit
ip default-gateway 192.168.99.1

# 4. Trunk Port Configurations & Security
# Fa0/1 connects to Router R1
interface FastEthernet0/1
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk native vlan 99
 switchport trunk allowed vlan 10,20,99
 no shutdown

# Fa0/2 connects to SW-Edge
interface FastEthernet0/2
 switchport mode trunk
 switchport trunk native vlan 99
 switchport trunk allowed vlan 10,20,99
 no shutdown

# 5. Switched Port Analyzer (SPAN) Configuration (Telemetry)
# Source: Trunk Fa0/1 (copies both incoming and outgoing traffic)
monitor session 1 source interface FastEthernet0/1 both
# Destination: Fa0/24 (monitors packet streams using Wireshark)
monitor session 1 destination interface FastEthernet0/24
exit
write
